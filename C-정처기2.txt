#include <stdio.h>
type struct{
	int num;
	int score;
} Score;
int main(){
	int x = 70;
	Score data[] = {1, 60, 2, 70, 3, 90, 4, 95, 5, 100}		
	int length = sizeof(data)/sizeof(data[0]);					
	int left = 0, right = length-1, mid;
	while(left <= right){								 		
		mid = (left + right) / 2;							
		if(x == data[mid].score)												
			break;																
		else if(x>data[mid].score)							
			left = mid + 1;									
		else												
			right = mid - 1;								
	}
	printf("%d%d\n", data[mid].num, data[mid].score);
		return 0;
}
