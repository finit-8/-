#include <stdio.h>
void swap(int *ptr1, int *ptr2){		
	int temp = *ptr1;					
	*ptr1 = *ptr2;						
	*ptr2 = temp;						
}
void func(int *arr, int n){				
	int *ptr1 = &arr[0];				
	int *ptr2 = &arr[n-1];				
	while(ptr1 < ptr2){					
		swap(ptr1, ptr2);			
		ptr1++;							
		ptr2--;							
	}
}
int main(){
	int arr[] = {1, 2, 3, 4, 5};
	func(arr, 5);
	printf("%d", arr[2]);
	return 0;
}
출력 : 
arr → 배열의 첫 번째 요소의 주소이고, 그 요소의 타입은 int*
&arr[0] -> 배열의 첫 번째 요소의 주소이고, 그 요소의 타입은 int*
&arr → 배열의 전체의 주소이고, 그 배열의 타입은 int (*)[5]
arr[0] → 값 (주소 아님 ❌)
